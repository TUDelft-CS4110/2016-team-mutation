	public final GL30 gl30;

	protected GL30Profiler (GL30 gl30) {
		this.gl30 = gl30;
	}
	
	private void check () {
		int error = gl30.glGetError();
		while (error != GL20.GL_NO_ERROR) {
			listener.onError(error);
			error = gl30.glGetError();
		}
	}
